<DB 쿼리 작성문법>
1. 테이블 생성
CREATE TABLE 테이블명(
  컬럼명 데이터타입 [제약조건],    -------> [제약조건]은 생략 갸능!
  컬럼명 데이터타입 [제약조건],
  컬럼명 데이터타입 [제약조건],
 );

* 제약조건
- PRIMARY KEY : 기본키, 각 행을 구분하는 식별자, 유일성과 최소성을 만족해야하며, NULL 데이터  X
- NOT NULL : NULL 데이터 들어갈 수 X
- UNIQUE : 중복데이터 X , NULL 데이터는 O
- DEFAULT 0 : 해당컬럼에 데이터 삽입 시 값을 명시하지 않으면 0을 삽입
- DEFAULT SYSDATE( ) : 데이터 타입이 DATETIME인 컬럼에 사용가능, 데이터 삽입시 명령어가 실행되는 시점의 날짜 및 시간을 
  자동으로 삽입

2. 데이터 삽입
* 기본문법 
INSERT INTO 테이블명 (
   컬럼명1, 컬럼명2
 ) VALUES(
    컬럼1에 맞는 데이터	, 컬럼2에 맞는 데이터
)

- 데이터 삽입 시 모든 컬럼 나열 X
- 모든 컬럼에 데이터를 추가한다면 : INSERT INTO 테이블명 VALUES (데이터1, 데이터 2, ....);
  -> 테이블 생성 시 작성한 컬럼명과 동일한 순서로 작성하면 됨!

3. 데이터 조회
* 기본 문법 : SELECT 조회할 컬럼명 FROM 테이블명;

* 조건 : WHERE
SELECT 조회할 컬럼명
  FROM 테이블명
 WHERE 조건

* 정렬 : ORDER BY
- ASC : 오름차순(기본값), DESC : 내림차순
- ORDER BY는 맨 마지막줄에 작성되고, 마지막에 해석됨.

SELECT 조회할 컬럼명
  FROM 테이블명
 WHERE 조건
 ORDER BY 컬럼명 정렬순

* 별칭 : AS or AS 생략

* AND 연산자 
WHERE SAL >= 300 AND JOB = '과장' => 급여가 300이상이고 직급이 과장인 사원

* BETWEEN A AND B 연산자
WHERE SAL BETWEEN 300 AND 800; => 급여가 300이상이면서 800이하인 사원

* OR 연산자
WHERE SAL <= 400 OR SAL >= 900; => 급여가 400 이하이거나 900이상인 사원

* IN 연산자
WHERE SAL IN (300, 400); => 급여가 300이거나 400인 사원
- 소괄호 안에는 필요한 만큼 데이터를 넣을 수 있음.

* != 연산자 : 아니다
WHERE JOB != '사원' => 직급이 사원이 아닌 사원

* NULL 
 1) IS NULL : NULL 인 데이터 조회
 2) IS NOT NULL : NULL이 아닌 데이터 조회

* LIKE 연산자 & 와일드카드(% , _ )
- ' % ' 는 0글자 이상의 어떤 글자를 의미 (글자수 제한 X)
- ' _ ' 는 어떤 한 글자를 의미 (글자수 제한 O)
-  '김%' : 김으로 시작하는 모든 글자 => 김가, 김가나, 김나다라.....
-  '김_' : 김으로 시작하는 2글자 => 김군
-  '_김_ _ _' : 2번째 글자가 김인 5글자
-  '_김%' : 2번째 글자가 김인 모든 글자
-  '%김__' : 마지막에서 3번째 글자가 김인 모든 글자(앞에는 글자 수 제한이 없기에 몇 번째 인지 모름)

ex) WHERE ENAME LIKE '%이%' => 사원명이 '이'와 일치하는 사원

* DISTINCT : 중복제거
- 기본 문법 : SELECT DISTINCT 컬럼명 FROM 테이블명;
 ex) SELECT DISTINCT DEPTNO FROM EMP; => EMP테이블에서 중복되는 부서번호를 제외한 부서번호를 조회
- 조회 쿼리문에서 한번만 사용! 각 컬럼 앞에 사용하면 오류 남
- 여러 컬럼에서 중복제거를 한다면 그 선택된 컬럼의 모든 데이터가 중복일 때 제거됨.

4. 데이터 수정
* 기본 문법 : 
UPDATE 테이블명 
      SET 컬럼명 = 변경할 값, 
            컬럼명 = 변경할 값,.... 
 WHERE 조건;

5. 데이터 삭제
* 기본 문법 : 
DELETE FROM 테이블명
WHERE 조건;


★ 쿼리 문의 해석 순서 ★
FROM -> WHERE -> GROUP BY -> HAVING -> SELECT -> ORDER BY
